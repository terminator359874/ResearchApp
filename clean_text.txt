            // Assert
            results.Should().Contain("Борщ");

using Xunit;
using FluentAssertions;
using RecipeSearchApp.Services;
namespace RecipeSearchApp.Tests.Services
{
    public class BloomFilterTests
    {
        [Fact]
        public void Add_SingleItem_ShouldBeDetectable()
        {
            // Arrange
            var filter = new BloomFilter(size: 1000, hashFunctionCount: 3);
            var item = "картофель";
            // Act
            filter.Add(item);
            // Assert
            filter.MightContain(item).Should().BeTrue();
        }
        [Fact]
        public void MightContain_NonExistentItem_ShouldReturnFalse()
        {
            // Arrange
            var filter = new BloomFilter();
            filter.Add("морковь");
            // Act
            var result = filter.MightContain("капуста");
            // Assert (может быть false, но не гарантированно)
            // Bloom фильтр не даёт ложноотрицательных результатов
            result.Should().BeFalse();
        }
        [Fact]
        public void Add_MultipleItems_AllShouldBeDetectable()
        {
            // Arrange
            var filter = new BloomFilter();
            var items = new[] { "лук", "чеснок", "перец", "соль" };
            // Act
            foreach (var item in items)
                filter.Add(item);
            // Assert
            foreach (var item in items)
                filter.MightContain(item).Should().BeTrue();
        }
        [Fact]
        public void MightContain_CaseInsensitive_ShouldWork()
        {
            // Arrange
            var filter = new BloomFilter();
            filter.Add("Томат");
            // Act &amp; Assert
            filter.MightContain("томат").Should().BeTrue();
            filter.MightContain("ТОМАТ").Should().BeTrue();
        }
        [Fact]
        public void Clear_AfterAdding_ShouldRemoveAll()
        {
            // Arrange
            var filter = new BloomFilter();
            filter.Add("сахар");
            filter.Add("соль");
            // Act
            filter.Clear();
            // Assert
            filter.MightContain("сахар").Should().BeFalse();
            filter.MightContain("соль").Should().BeFalse();
        }
        [Fact]
        public void FalsePositiveRate_ShouldBeLow()
        {
            // Arrange
            var filter = new BloomFilter(size: 10000, hashFunctionCount: 3);
            var testItems = new[] { "item1", "item2", "item3", "item4", "item5" };
            foreach (var item in testItems)
                filter.Add(item);
            // Act - проверяем 100 случайных слов, которых нет
            var falsePositives = 0;
            for (int i = 0; i &lt; 100; i++)
            {
                var testWord = $"nonexistent_{i}_{Guid.NewGuid()}";
                if (filter.MightContain(testWord))
                    falsePositives++;
            }
            // Assert - вероятность ложноположительных &lt; 10%
            falsePositives.Should().BeLessThan(10);
        }
    }
}

using Xunit;
using FluentAssertions;
using RecipeSearchApp.Models;
using RecipeSearchApp.Services;
namespace RecipeSearchApp.Tests.Services
{
    public class InMemoryCacheTests
    {
        [Fact]
        public void Set_And_Get_ShouldReturnSameRecipe()
        {
            // Arrange
            var cache = new InMemoryCache();
            var recipe = new Recipe
            {
                Id = 1,
                Title = "Тестовый рецепт",
                Ingredients = "тест",
                Instructions = "тест",
                Category = "Тесты",
                PrepTimeMinutes = 10
            };
            // Act
            cache.Set(1, recipe);
            var retrieved = cache.Get(1);
            // Assert
            retrieved.Should().NotBeNull();
            retrieved!.Title.Should().Be("Тестовый рецепт");
        }
        [Fact]
        public void Get_NonExistentKey_ShouldReturnNull()
        {
            // Arrange
            var cache = new InMemoryCache();
            // Act
            var result = cache.Get(999);
            // Assert
            result.Should().BeNull();
        }
        [Fact]
        public void Set_ExceedingMaxSize_ShouldEvictOldest()
        {
            // Arrange
            var cache = new InMemoryCache(maxSize: 3);
            for (int i = 1; i &lt;= 4; i++)
            {
                cache.Set(i, new Recipe { Id = i, Title = $"Recipe {i}" });
            }
            // Act
            var oldest = cache.Get(1); // Должен быть вытеснен
            var newest = cache.Get(4);
            // Assert
            oldest.Should().BeNull();
            newest.Should().NotBeNull();
        }
        [Fact]
        public void Get_UpdatesLastAccessTime()
        {
            // Arrange
            var cache = new InMemoryCache(maxSize: 2);
            cache.Set(1, new Recipe { Id = 1, Title = "First" });
            cache.Set(2, new Recipe { Id = 2, Title = "Second" });
            // Act
            cache.Get(1); // Обновляет LastAccessTime для ключа 1
            cache.Set(3, new Recipe { Id = 3, Title = "Third" });
            // Assert
            cache.Get(1).Should().NotBeNull(); // Не вытеснен
            cache.Get(2).Should().BeNull();    // Вытеснен как самый старый
        }
        [Fact]
        public void Clear_ShouldRemoveAllEntries()
        {
            // Arrange
            var cache = new InMemoryCache();
            cache.Set(1, new Recipe { Id = 1, Title = "Test" });
            cache.Set(2, new Recipe { Id = 2, Title = "Test2" });
            // Act
            cache.Clear();
            // Assert
            cache.Get(1).Should().BeNull();
            cache.Get(2).Should().BeNull();
        }
        [Fact]
        public void GetStatistics_ShouldReturnCorrectData()
        {
            // Arrange
            var cache = new InMemoryCache();
            cache.Set(1, new Recipe { Id = 1, Title = "Test" });
            // Act
            cache.Get(1);
            cache.Get(1);
            var stats = cache.GetStatistics();
            // Assert
            stats["CachedItems"].Should().Be(1);
            stats["TotalHits"].Should().Be(2);
        }
        [Fact]
        public async Task Get_ExpiredEntry_ShouldReturnNull()
        {
            // Arrange
            var cache = new InMemoryCache(expirationMinutes: 0); // 0 минут = мгновенное истечение
            cache.Set(1, new Recipe { Id = 1, Title = "Test" });
            // Act
            await Task.Delay(100); // Небольшая задержка
            var result = cache.Get(1);
            // Assert
            result.Should().BeNull();
        }
    }

[Fact]

    public int PrepTimeMinutes { get; set; }
    public DateTime CreatedAt { get; set; } = DateTime.Now;
    public int ViewCount { get; set; }
    // Новые поля для рейтинга
    public double AverageRating { get; set; }

// Шаг 3: Реализуем метод AddRatingAsync в DatabaseService (Green)
public async Task AddRatingAsync(int recipeId, int rating)
{
    await Task.Run(() =&gt;
    {
        using var connection = new SqliteConnection($"Data Source={_dbPath}");
        connection.Open();
        // Получаем текущие значения
        using var selectCmd = connection.CreateCommand();
        selectCmd.CommandText =
            "SELECT AverageRating, RatingCount FROM Recipes WHERE Id = @id";
        selectCmd.Parameters.AddWithValue("@id", recipeId);
        using var reader = selectCmd.ExecuteReader();
        if (!reader.Read()) return;
        var currentAvg = reader.GetDouble(0);
        var currentCount = reader.GetInt32(1);
        reader.Close();
        // Вычисляем новый средний рейтинг
        var newCount = currentCount + 1;
        var newAvg = ((currentAvg * currentCount) + rating) / newCount;
        // Обновляем
        using var updateCmd = connection.CreateCommand();
        updateCmd.CommandText = @"
            UPDATE Recipes
            SET AverageRating = @avg, RatingCount = @count
            WHERE Id = @id";
        updateCmd.Parameters.AddWithValue("@avg", newAvg);
        updateCmd.Parameters.AddWithValue("@count", newCount);
        updateCmd.Parameters.AddWithValue("@id", recipeId);
        updateCmd.ExecuteNonQuery();
    });
}
